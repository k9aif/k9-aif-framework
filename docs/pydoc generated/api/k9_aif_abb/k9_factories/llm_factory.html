 13    """
 14    K9-AIF Factory ABB - LLMFactory
 15    -------------------------------
 16    Unified model manager for Granite / Ollama / watsonx.ai models.
 17
 18    Backward-compatible with older agents expecting:
 19      - LLMFactory.is_bootstrapped
 20      - LLMFactory._models
 21      - LLMFactory.get_model()
 22    """
 23
 24    # ------------------------------------------------------------------
 25    # Internal state
 26    # ------------------------------------------------------------------
 27    _bootstrapped: bool = False
 28    _cfg: Dict[str, Any] = {}
 29    _instances: Dict[str, Any] = {}
 30
 31    # Legacy compatibility (for AdvisorAgent, etc.)
 32    _models: Dict[str, str] = {}
 33    _is_bootstrapped: bool = False  # internal flag for callable property
 34
 35    # ------------------------------------------------------------------
 36    # Callable compatibility shim
 37    # ------------------------------------------------------------------
 38    @classmethod
 39    def is_bootstrapped(cls) -> bool:
 40        """Legacy callable compatibility (for old agents)."""
 41        return cls._bootstrapped or cls._is_bootstrapped
 42
 43    # ------------------------------------------------------------------
 44    # Bootstrap
 45    # ------------------------------------------------------------------
 46    @classmethod
 47    def bootstrap(cls, config):
 48        """Initialize the LLM factory (supports ABB + SBB flattened configs)."""
 49        try:
 50            # Normalize structure for flattened SBB configs
 51            if "inference" not in config and "llm_factory" in config:
 52                print("[LLMFactory] [INFO] Normalizing flattened config -> nesting under inference.llm_factory")
 53                config = {"inference": {"llm_factory": config["llm_factory"]}}
 54
 55            llm_cfg = config.get("inference", {}).get("llm_factory", {})
 56            if not llm_cfg:
 57                raise ValueError("[LLMFactory] [ERROR] Missing inference.llm_factory section in config")
 58
 59            # [OK] FIX: Store the entire config in cls._cfg
 60            cls._cfg = config
 61            cls._bootstrapped = True
 62            cls._is_bootstrapped = True
 63            
 64            # [OK] FIX: Populate cls._models for legacy compatibility
 65            cls._models = llm_cfg.get("models", {})
 66            
 67            # Store configuration attributes
 68            cls.backend = llm_cfg.get("backend", "ollama")
 69            cls.provider = llm_cfg.get("provider", "ollama")
 70            cls.base_url = llm_cfg.get("base_url", "")
 71
 72            print(f"[LLMFactory] [INFO] Loaded models -> {cls._models}")
 73            print(f"[LLMFactory] [INFO] Base URL -> {cls.base_url}")
 74            print(f"[LLMFactory] [OK] Provider -> {cls.provider}")
 75            print(f"[LLMFactory] [OK] Bootstrap complete")
 76
 77        except Exception as e:
 78            print(f"[LLMFactory] [ERROR] Bootstrap failed: {e}")
 79            traceback.print_exc()
 80            raise
 81             
 82    # ------------------------------------------------------------------
 83    # Get
 84    # ------------------------------------------------------------------
 85    @classmethod
 86    def get(cls, alias: Optional[str] = "general") -> OllamaLLM:
 87        """
 88        Return a cached or newly constructed OllamaLLM instance for the alias.
 89        Defensive default ensures alias='general' if None is passed.
 90        """
 91        alias = alias or "general"  # safeguard for agents passing None
 92
 93        if alias in cls._instances:
 94            return cls._instances[alias]
 95
 96        if not cls._bootstrapped:
 97            raise RuntimeError("LLMFactory.get() called before bootstrap().")
 98
 99        log = logging.getLogger("LLMFactory")
100
101        inf = (cls._cfg or {}).get("inference", {})
102        fcfg = inf.get("llm_factory") or {}
103        base_url = fcfg.get("base_url", "http://localhost:11434")
104        model_id = (fcfg.get("models") or {}).get(alias)
105
106        if not model_id:
107            raise KeyError(
108                f"LLM model alias '{alias}' is not configured under inference.llm_factory.models"
109            )
110
111        # Quiet initialization (no announce)
112        inst = OllamaLLM(host=base_url, model=model_id)
113        cls._instances[alias] = inst
114        log.info(f"LLM instance ready [{alias} -> {model_id}] (cached).")
115
116        return inst
117
118    # ------------------------------------------------------------------
119    # Legacy accessor (for older agents)
120    # ------------------------------------------------------------------
121    @classmethod
122    def get_model(cls, name: str = "general") -> Optional[str]:
123        """Return model identifier by logical name (legacy use)."""
124        if not cls.is_bootstrapped():
125            logging.getLogger("LLMFactory").warning(
126                "get_model() called before bootstrap"
127            )
128        return cls._models.get(name)
129
130    # ------------------------------------------------------------------
131    # Reset (for tests or reloads)
132    # ------------------------------------------------------------------
133    @classmethod
134    def reset(cls):
135        """Clear all loaded models and reset bootstrap flag."""
136        cls._instances.clear()
137        cls._bootstrapped = False
138        cls._is_bootstrapped = False
139        cls._cfg = {}
140        cls._models = {}
141        logging.getLogger("LLMFactory").info("LLMFactory reset complete.")
