 284def verify():
 285    """Run a framework smoke test — no LLM needed."""
 286    print("K9-AIF Framework — Smoke Test")
 287    print("=" * 40)
 288
 289    passed = 0
 290    failed = 0
 291
 292    def check(label, fn):
 293        nonlocal passed, failed
 294        try:
 295            fn()
 296            print(f"  ✓  {label}")
 297            passed += 1
 298        except Exception as exc:
 299            print(f"  ✗  {label}: {exc}")
 300            failed += 1
 301
 302    check("BaseAgent import", lambda: __import__(
 303        "k9_aif_abb.k9_core.agent.base_agent", fromlist=["BaseAgent"]))
 304    check("BaseRouter import", lambda: __import__(

 306    check("BaseOrchestrator import", lambda: __import__(
 307        "k9_aif_abb.k9_core.orchestration.base_orchestrator", fromlist=["BaseOrchestrator"]))

 309        "k9_aif_abb.k9_squad.base_squad", fromlist=["BaseSquad"]))
 310    check("LLMFactory import", lambda: __import__(
 311        "k9_aif_abb.k9_factories.llm_factory", fromlist=["LLMFactory"]))
 312    check("ModelRouterFactory import", lambda: __import__(
 313        "k9_aif_abb.k9_factories.model_router_factory", fromlist=["ModelRouterFactory"]))

 315        "k9_aif_abb.k9_agents.registry.agent_registry", fromlist=["AgentRegistry"]))
 316    check("Governance pipeline import", lambda: __import__(
 317        "k9_aif_abb.k9_core.governance.pipeline", fromlist=["require_governance"]))
 318    check("InferenceRequest import", lambda: __import__(
 319        "k9_aif_abb.k9_inference.models.inference_request", fromlist=["InferenceRequest"]))
 320    check("K9CrewAIAdapter import", lambda: __import__(
 321        "k9_aif_abb.k9_adapters.crewai.k9_crewai_adapter", fromlist=["K9CrewAIAdapter"]))
 322    check("BaseValidationLoopAgent import", lambda: __import__(
 323        "k9_aif_abb.k9_agents.validation", fromlist=["BaseValidationLoopAgent"]))

 325    def _test_governance():
 326        from k9_aif_abb.k9_core.governance.pipeline import require_governance, NoopGovernance
 327        gov = require_governance(None, "development")
 328        assert isinstance(gov, NoopGovernance)
 329    check("Governance enforcement (dev mode)", _test_governance)

 331    def _test_registry():
 332        from k9_aif_abb.k9_agents.registry.agent_registry import AgentRegistry
 333        from k9_aif_abb.k9_core.agent.base_agent import BaseAgent
 334
 335        class _TestAgent(BaseAgent):

 338                return {"ok": True}

 342        agent = registry.create("_TestAgent")
 343        result = agent.execute({})
 344        assert result["ok"] is True
 345    check("Agent registry — register + create + execute", _test_registry)
 346
 347    print()
 348    print(f"Results: {passed} passed, {failed} failed")
 349
 350    if failed == 0:
 351        print()
 352        print("K9-AIF is installed correctly and ready to use.")
 353        print("Run 'k9aif --help develop' to start building.")
 354    else:
 355        print()
 356        print("Some checks failed. Ensure Python 3.11+ and retry.")

 686# K9-AIF Context — Architecture-First Agentic AI Framework

 688K9-AIF is a Python framework for designing and building governed, modular multi-agent AI systems.

 691
 692## Core Concepts
 693
 694**ABB (Architecture Building Block)** — Abstract contracts in `k9_aif_abb/`. Define interfaces,
 695lifecycle, and governance hooks. Never contain domain logic.
 696

 699
 700## Execution Hierarchy
 701
 702```
 703Event → Router
 704    └── Orchestrator
 705            └── Squad
 706                    └── Agent → LLM
 707```

 718| Router | Orchestrators only | Squads, Agents |

1562def _scan_subpackage(subpackage: str) -> list[tuple[str, str]]:
1563    """Dynamically discover classes in a k9_aif_abb subpackage."""
1564    import pkgutil
1565    import importlib
1566    import inspect
1567
1568    results = []
1569    try:
1570        pkg = importlib.import_module(f"k9_aif_abb.{subpackage}")
1571        prefix = f"k9_aif_abb.{subpackage}."
1572        pkg_path = getattr(pkg, "__path__", [])
1573
1574        for finder, mod_name, _ in pkgutil.walk_packages(pkg_path, prefix=prefix):
1575            try:
1576                mod = importlib.import_module(mod_name)
1577                for cls_name, cls in inspect.getmembers(mod, inspect.isclass):
1578                    if cls.__module__ == mod_name and not cls_name.startswith("_"):
1579                        results.append((cls_name, mod_name))
1580            except Exception:
1581                pass
1582    except Exception:
1583        pass
1584
1585    return sorted(set(results))
