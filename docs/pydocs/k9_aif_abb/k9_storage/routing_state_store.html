126    def ensure_session(self, session_id: str, user_id: str) -> None:
127        with self.db.get_session() as session:
128            stmt = select(self.sessions.c.session_id).where(
129                self.sessions.c.session_id == session_id
130            )
131            exists = session.execute(stmt).fetchone()
132
133            if not exists:
134                session.execute(
135                    insert(self.sessions).values(
136                        session_id=session_id,
137                        user_id=user_id,
138                        created_at=datetime.utcnow(),
139                        last_active=datetime.utcnow(),
140                        status="ACTIVE",
141                    )
142                )
143            else:
144                session.execute(
145                    update(self.sessions)
146                    .where(self.sessions.c.session_id == session_id)
147                    .values(last_active=datetime.utcnow())
148                )
149
150            session.commit()

191        self,
192        session_id: str,
193        role: str,
194        content: str,
195        token_count: Optional[int] = None,
196        compressed_flag: bool = False,
197    ) -> int:
198        with self.db.get_session() as session:
199            stmt = (
200                select(self.session_turns.c.turn_index)
201                .where(self.session_turns.c.session_id == session_id)
202                .order_by(self.session_turns.c.turn_index.desc())
203            )
204
205            last = session.execute(stmt).fetchone()
206            next_index = (last[0] + 1) if last else 1
207
208            result = session.execute(
209                insert(self.session_turns).values(
210                    session_id=session_id,
211                    turn_index=next_index,
212                    role=role,
213                    content=content,
214                    token_count=token_count,
215                    compressed_flag=compressed_flag,
216                    created_at=datetime.utcnow(),
217                )
218            )
219
220            session.commit()
221
222            inserted_turn_id = result.inserted_primary_key[0]
223            return inserted_turn_id
