k9_aif_abb.k9_security.adapters

k9_security.adapters — Secret manager adapter implementations.

Available adapters::

env    → EnvSecretAdapter    (default — reads os.environ)
vault  → VaultSecretAdapter  (HashiCorp Vault via VAULT_ADDR + VAULT_TOKEN)
aws    → AwsSecretAdapter    (AWS Secrets Manager via boto3)
ibm    → IbmSecretAdapter    (IBM Secrets Manager via ibm-platform-services)

Typical import::

from k9_aif_abb.k9_security.adapters.env_adapter import EnvSecretAdapter

class VaultSecretAdapter(k9_aif_abb.k9_core.security.base_secret_manager.BaseSecretManager): 
42class VaultSecretAdapter(BaseSecretManager):
43    """Retrieves secrets from HashiCorp Vault KV v2."""
44
45    def __init__(self, config: Optional[Dict[str, Any]] = None):
46        self._config = config or {}
47        cfg = self._config.get("secrets", self._config)
48
49        # Credentials come from env, never from config.yaml
50        self._addr   = os.environ.get("VAULT_ADDR",  cfg.get("vault_addr", ""))
51        self._token  = os.environ.get("VAULT_TOKEN", cfg.get("vault_token", ""))
52        self._mount  = cfg.get("vault_mount", "secret")
53        self._prefix = cfg.get("vault_path_prefix", "")
54        self._client = None
55
56    def _ensure_client(self):
57        if self._client is not None:
58            return
59        try:
60            import hvac  # type: ignore
61            self._client = hvac.Client(url=self._addr, token=self._token)
62            if not self._client.is_authenticated():
63                raise RuntimeError("Vault authentication failed — check VAULT_TOKEN")
64            log.info("[VaultSecretAdapter] authenticated at %s", self._addr)
65        except ImportError as exc:
66            raise RuntimeError(
67                "hvac package required for VaultSecretAdapter: pip install hvac"
68            ) from exc
69
70    def _resolve_path(self, key: str) -> str:
71        return f"{self._prefix}{key}" if self._prefix else key
72
73    def get(self, key: str) -> str:
74        self._ensure_client()
75        path = self._resolve_path(key)
76        try:
77            resp = self._client.secrets.kv.v2.read_secret_version(
78                mount_point=self._mount, path=path
79            )
80            value = resp["data"]["data"].get(key)
81            if value is None:
82                raise KeyError(key)
83            return str(value)
84        except Exception as exc:
85            # Normalise all Vault errors to KeyError so callers handle uniformly
86            if isinstance(exc, KeyError):
87                raise
88            log.warning("[VaultSecretAdapter] failed to retrieve %s: %s", key, exc)
89            raise KeyError(key) from exc

39class AwsSecretAdapter(BaseSecretManager):
40    """Retrieves secrets from AWS Secrets Manager."""
41
42    def __init__(self, config: Optional[Dict[str, Any]] = None):
43        self._config = config or {}
44        cfg = self._config.get("secrets", self._config)
45        self._region = cfg.get("aws_region") or os.environ.get("AWS_DEFAULT_REGION", "us-east-1")
46        self._client = None
47
48    def _ensure_client(self):
49        if self._client is not None:
50            return
51        try:
52            import boto3  # type: ignore
53            self._client = boto3.client("secretsmanager", region_name=self._region)
54            log.info("[AwsSecretAdapter] client initialised (region=%s)", self._region)
55        except ImportError as exc:
56            raise RuntimeError(
57                "boto3 package required for AwsSecretAdapter: pip install boto3"
58            ) from exc
59
60    def _resolve_name(self, key: str) -> str:
61        return key
62
63    def get(self, key: str) -> str:
64        self._ensure_client()
65        name = self._resolve_name(key)
66        try:
67            resp = self._client.get_secret_value(SecretId=name)
68            raw = resp.get("SecretString") or resp.get("SecretBinary", b"").decode()
69            # If the secret is a JSON object, try extracting the key directly
70            try:
71                parsed = json.loads(raw)
72                if isinstance(parsed, dict):
73                    value = parsed.get(key)
74                    if value is not None:
75                        return str(value)
76                    # Return the whole JSON string if key not found in object
77                    return raw
78            except (json.JSONDecodeError, TypeError):
79                pass
80            if not raw:
81                raise KeyError(key)
82            return raw
83        except KeyError:
84            raise
85        except Exception as exc:
86            log.warning("[AwsSecretAdapter] failed to retrieve %s: %s", key, exc)
87            raise KeyError(key) from exc

class IbmSecretAdapter(k9_aif_abb.k9_core.security.base_secret_manager.BaseSecretManager): 
39class IbmSecretAdapter(BaseSecretManager):
40    """Retrieves secrets from IBM Secrets Manager."""
41
42    def __init__(self, config: Optional[Dict[str, Any]] = None):
43        self._config = config or {}
44        cfg = self._config.get("secrets", self._config)
45        self._endpoint = cfg.get("ibm_sm_endpoint", "")
46        self._group    = cfg.get("ibm_sm_secret_group", "default")
47        self._api_key  = os.environ.get("IBM_CLOUD_API_KEY", "")
48        self._client   = None
49
50    def _ensure_client(self):
51        if self._client is not None:
52            return
53        try:
54            from ibm_platform_services import SecretsManagerV2  # type: ignore
55            from ibm_cloud_sdk_core.authenticators import IAMAuthenticator  # type: ignore
56            if not self._api_key:
57                raise RuntimeError("IBM_CLOUD_API_KEY environment variable not set")
58            auth = IAMAuthenticator(self._api_key)
59            self._client = SecretsManagerV2(authenticator=auth)
60            self._client.set_service_url(self._endpoint)
61            log.info("[IbmSecretAdapter] client initialised (endpoint=%s)", self._endpoint)
62        except ImportError as exc:
63            raise RuntimeError(
64                "ibm-platform-services and ibm-cloud-sdk-core required: "
65                "pip install ibm-platform-services ibm-cloud-sdk-core"
66            ) from exc
67
68    def _resolve_name(self, key: str) -> str:
69        return key
70
71    def get(self, key: str) -> str:
72        self._ensure_client()
73        name = self._resolve_name(key)
74        try:
75            resp = self._client.list_secrets(groups=[self._group]).get_result()
76            for secret in resp.get("secrets", []):
77                if secret.get("name") == name:
78                    secret_id = secret["id"]
79                    detail = self._client.get_secret(id=secret_id).get_result()
80                    value = (
81                        detail.get("payload")
82                        or detail.get("secret_data", {}).get("payload", "")
83                    )
84                    if not value:
85                        raise KeyError(key)
86                    return str(value)
87            raise KeyError(key)
88        except KeyError:
89            raise
90        except Exception as exc:
91            log.warning("[IbmSecretAdapter] failed to retrieve %s: %s", key, exc)
92            raise KeyError(key) from exc
