 34class RealGovernance:
 35    """Minimal concrete governance used across tests."""
 36    def pre_process(self, payload: dict, ctx=None) -> dict:
 37        payload["_governed"] = True
 38        return payload
 39
 40    def post_process(self, payload: dict, ctx=None) -> dict:
 41        payload["_post_governed"] = True
 42        return payload
