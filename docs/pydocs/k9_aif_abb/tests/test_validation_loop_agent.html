 24class ConvergingAgent(BaseValidationLoopAgent):
 25    """
 26    Converges toward a target value over iterations.
 27    Guess = current iteration number. Finalizes when guess == target.
 28    """
 29
 30    layer = "ConvergingAgent"
 31
 32    def generate_hypothesis(self, loop_ctx: ValidationLoopContext):
 33        return loop_ctx.iteration
 34
 35    def run_validation(self, hypothesis, loop_ctx: ValidationLoopContext):
 36        target = loop_ctx.payload.get("target", 3)
 37        return {"correct": hypothesis == target}
 38
 39    def evaluate_observation(self, tool_result, loop_ctx: ValidationLoopContext):
 40        return {
 41            "correct":    tool_result["correct"],
 42            "confidence": 1.0 if tool_result["correct"] else 0.3,
 43        }
 44
 45    def should_continue(self, observation, loop_ctx: ValidationLoopContext):
 46        if observation["correct"]:
 47            return ValidationDisposition.FINALIZE
 48        return ValidationDisposition.CONTINUE
 49
 50    def finalize(self, loop_ctx: ValidationLoopContext) -> ValidationLoopResult:
 51        last = loop_ctx.steps[-1] if loop_ctx.steps else None
 52        return ValidationLoopResult(
 53            disposition      = ValidationDisposition.FINALIZE,
 54            output           = {"validated": True, "guess": last.hypothesis if last else None},
 55            steps            = loop_ctx.steps,
 56            iterations       = loop_ctx.iteration,
 57            final_confidence = 1.0,
 58            evidence         = ["hypothesis matched target"],
 59        )

106class NeverConvergesAgent(BaseValidationLoopAgent):
107    """Always returns CONTINUE — hits max_iterations every time."""
108
109    layer = "NeverConvergesAgent"
110
111    def generate_hypothesis(self, loop_ctx): return "h"
112    def run_validation(self, h, loop_ctx):   return {}
113
114    def evaluate_observation(self, tool_result, loop_ctx):
115        return {"confidence": 0.2}
116
117    def should_continue(self, observation, loop_ctx):
118        return ValidationDisposition.CONTINUE
119
120    def finalize(self, loop_ctx) -> ValidationLoopResult:
121        return ValidationLoopResult(
122            disposition      = ValidationDisposition.FINALIZE,
123            output           = {"reason": "best_effort"},
124            steps            = loop_ctx.steps,
125            iterations       = loop_ctx.iteration,
126            final_confidence = self._last_confidence(loop_ctx),
