 31    def __init__(self, name: str = "BasePersistence", monitor=None,
 32                 config: Optional[Dict[str, Any]] = None):
 33        super().__init__(monitor=monitor)
 34        self.name = name
 35        self.config = config or {}
 36        self.logger = logging.getLogger(self.name)

 39        # Lazy import to avoid circular dependency with factories
 40        # ------------------------------------------------------------------
 41        try:
 42            from k9_aif_abb.k9_factories.message_factory import MessageFactory
 43            self.message_bus = MessageFactory.get_global() or MessageFactory.create(self.config)
 44            if self.message_bus:
 45                self.logger.debug(f"[{self.name}] Attached global MessageBus")
 46        except Exception as e:
 47            self.logger.debug(f"[{self.name}] MessageBus unavailable: {e}")
 48            self.message_bus = None

 51    async def log(self, message: str, level: str = "INFO"):
 52        """Layer-aware async/sync log; writes to Python logger, monitor, and MessageBus."""
 53        try:
 54            level = level.upper()
 55            formatted = f"[{self.layer}:{self.name}] {message}"
 56            getattr(self.logger, level.lower(), self.logger.info)(formatted)
 57
 58            # Forward to BaseComponent's log if available
 59            if hasattr(super(), "log"):
 60                await super().log(message, level)
 61
 62            # Emit structured event if bus exists
 63            if self.message_bus:
 64                event = {
 65                    "timestamp": time.strftime("%Y-%m-%dT%H:%M:%S"),
 66                    "component": self.name,
 67                    "layer": self.layer,
 68                    "level": level,
 69                    "message": message,
 70                    "topic": self.config.get("messaging", {}).get("topic", "k9aif-events"),
 71                }
 72
 73                if asyncio.iscoroutinefunction(self.message_bus.publish):
 74                    loop = asyncio.get_event_loop()
 75                    if loop.is_running():
 76                        asyncio.create_task(self.message_bus.publish(event))
 77                    else:
 78                        await self.message_bus.publish(event)
 79                else:
