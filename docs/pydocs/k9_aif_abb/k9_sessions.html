14class InMemorySessionStore(BaseSessionStore):
15    """
16    Thread-safe in-memory session store.
17
18    Default store — zero dependencies. State is lost on restart.
19    Use for development, testing, and single-process deployments.
20    """
21
22    def __init__(self, config: Optional[Dict[str, Any]] = None) -> None:
23        self._store: Dict[str, K9Session] = {}
24        self._lock = Lock()
25
26    def get(self, session_id: str) -> Optional[K9Session]:
27        with self._lock:
28            return self._store.get(session_id)
29
30    def save(self, session: K9Session) -> None:
31        with self._lock:
32            self._store[session.session_id] = session
33
34    def delete(self, session_id: str) -> None:
35        with self._lock:
36            self._store.pop(session_id, None)
37
38    def exists(self, session_id: str) -> bool:
39        with self._lock:
40            return session_id in self._store
41
42    def list_active(self) -> List[K9Session]:
43        with self._lock:
44            return [
45                s for s in self._store.values()
46                if s.status == SessionStatus.ACTIVE
47            ]

class RedisSessionStore(k9_aif_abb.k9_core.session.base_session_store.BaseSessionStore): 
17class RedisSessionStore(BaseSessionStore):
18    """
19    Redis-backed session store for distributed / multi-process deployments.
20
21    Requires: pip install redis
22
23    Config keys (under session.redis):
24        host: localhost
25        port: 6379
26        db: 0
27        key_prefix: k9aif:session:
28    """
29
30    def __init__(self, config: Optional[Dict[str, Any]] = None) -> None:
31        self._config = (config or {}).get("session", {}).get("redis", {})
32        self._prefix = self._config.get("key_prefix", "k9aif:session:")
33        self._client = None  # lazy init
34
35    def _ensure_client(self):
36        if self._client is not None:
37            return
38        try:
39            import redis  # type: ignore
40            self._client = redis.Redis(
41                host=self._config.get("host", "localhost"),
42                port=int(self._config.get("port", 6379)),
43                db=int(self._config.get("db", 0)),
44                decode_responses=True,
45            )
46        except ImportError as exc:
47            raise RuntimeError("pip install redis required for RedisSessionStore") from exc
48
49    def _key(self, session_id: str) -> str:
50        return f"{self._prefix}{session_id}"
51
52    def get(self, session_id: str) -> Optional[K9Session]:
53        self._ensure_client()
54        raw = self._client.get(self._key(session_id))
55        if raw is None:
56            return None
57        return K9Session.from_dict(json.loads(raw))
58
59    def save(self, session: K9Session) -> None:
60        self._ensure_client()
61        self._client.setex(
62            self._key(session.session_id),
63            session.ttl,
64            json.dumps(session.to_dict()),
65        )
66
67    def delete(self, session_id: str) -> None:
68        self._ensure_client()
69        self._client.delete(self._key(session_id))
70
71    def exists(self, session_id: str) -> bool:
72        self._ensure_client()
73        return bool(self._client.exists(self._key(session_id)))
74
75    def list_active(self) -> List[K9Session]:
76        # Redis SCAN is expensive — not supported for list_active at scale
77        return []

class SQLiteSessionStore(k9_aif_abb.k9_core.session.base_session_store.BaseSessionStore): 
 33class SQLiteSessionStore(BaseSessionStore):
 34    """
 35    SQLite-backed session store — survives process restarts.
 36
 37    Config keys (under session.sqlite):
 38        db_path: /tmp/k9_sessions.db
 39    """
 40
 41    def __init__(self, config: Optional[Dict[str, Any]] = None) -> None:
 42        cfg = (config or {}).get("session", {}).get("sqlite", {})
 43        self._db_path = cfg.get("db_path", "/tmp/k9_sessions.db")
 44        self._lock = Lock()
 45        self._init_db()
 46
 47    def _init_db(self) -> None:
 48        with sqlite3.connect(self._db_path) as conn:
 49            conn.execute(_CREATE_TABLE)
 50            conn.commit()
 51
 52    def get(self, session_id: str) -> Optional[K9Session]:
 53        with self._lock:
 54            with sqlite3.connect(self._db_path) as conn:
 55                row = conn.execute(
 56                    "SELECT * FROM k9_sessions WHERE session_id = ?", (session_id,)
 57                ).fetchone()
 58        if row is None:
 59            return None
 60        return self._row_to_session(row)
 61
 62    def save(self, session: K9Session) -> None:
 63        d = session.to_dict()
 64        with self._lock:
 65            with sqlite3.connect(self._db_path) as conn:
 66                conn.execute(
 67                    """INSERT OR REPLACE INTO k9_sessions
 68                       (session_id, user_id, created_at, last_active,
 69                        ttl, context, metadata, status)
 70                       VALUES (?,?,?,?,?,?,?,?)""",
 71                    (
 72                        d["session_id"], d["user_id"],
 73                        d["created_at"], d["last_active"],
 74                        d["ttl"],
 75                        json.dumps(d["context"]),
 76                        json.dumps(d["metadata"]),
 77                        d["status"],
 78                    ),
 79                )
 80                conn.commit()
 81
 82    def delete(self, session_id: str) -> None:
 83        with self._lock:
 84            with sqlite3.connect(self._db_path) as conn:
 85                conn.execute(
 86                    "DELETE FROM k9_sessions WHERE session_id = ?", (session_id,)
 87                )
 88                conn.commit()
 89
 90    def exists(self, session_id: str) -> bool:
 91        with self._lock:
 92            with sqlite3.connect(self._db_path) as conn:
 93                row = conn.execute(
 94                    "SELECT 1 FROM k9_sessions WHERE session_id = ?", (session_id,)
 95                ).fetchone()
 96        return row is not None
 97
 98    def list_active(self) -> List[K9Session]:
 99        with self._lock:
100            with sqlite3.connect(self._db_path) as conn:
101                rows = conn.execute(
102                    "SELECT * FROM k9_sessions WHERE status = 'active'"
103                ).fetchall()
104        return [self._row_to_session(r) for r in rows]
105
106    @staticmethod
107    def _row_to_session(row) -> K9Session:
108        return K9Session.from_dict({
109            "session_id":  row[0],
110            "user_id":     row[1],
111            "created_at":  row[2],
112            "last_active": row[3],
113            "ttl":         row[4],
114            "context":     json.loads(row[5]),
115            "metadata":    json.loads(row[6]),
116            "status":      row[7],
117        })
