107        role = self.config.get("role", _DEFAULT_ROLE)
108        goal = self.config.get("goal", _DEFAULT_GOAL)

142            f"Iteration {loop_ctx.iteration}.\n"
143            f"Return a JSON object with exactly these keys:\n"
144            f"  conclusion      : string  — your finding for this iteration\n"

147            f"  remaining_steps : array of strings — the updated plan after this "
148            f"step; an empty array signals the task is complete\n"
149            f"  notes           : object  — scratchpad to carry forward to the "
150            f"next iteration (merge with, don't discard, prior notes)\n"

153    def run_validation(self, hypothesis: str, loop_ctx: ValidationLoopContext) -> str:
154        req = InferenceRequest(
155            prompt=hypothesis,
156            task_type=self.config.get("model", "reasoning"),
157            metadata={"agent": self.layer, "iteration": loop_ctx.iteration},

159        resp = llm_invoke(self.config, req)

162    def evaluate_observation(self, tool_result: str, loop_ctx: ValidationLoopContext) -> Dict[str, Any]:
163        data = self._parse_llm_json(tool_result)
164
165        plan_provided = isinstance(data.get("remaining_steps"), list)
166        if plan_provided:
167            loop_ctx.remaining_steps = [str(s) for s in data["remaining_steps"]]
168
169        notes = data.get("notes")
170        if isinstance(notes, dict):
171            loop_ctx.notes = notes
172
173        return {
174            "conclusion":     data.get("conclusion", str(tool_result)[:200]),
175            "confidence":     data.get("confidence", 0.5),
176            "reasoning":      data.get("reasoning", ""),
177            "plan_complete":  plan_provided and not loop_ctx.remaining_steps,
178        }

180    def should_continue(self, observation: Dict[str, Any], loop_ctx: ValidationLoopContext) -> ValidationDisposition:
181        threshold  = float(self.config.get("confidence_threshold", 0.8))
182        max_iter   = int(self.config.get("max_iterations", 5))
183        confidence = observation["confidence"]
184
185        if observation.get("plan_complete") or confidence >= threshold:
186            return ValidationDisposition.FINALIZE
187
188        if loop_ctx.iteration < max_iter - 1:
189            return ValidationDisposition.CONTINUE
190
191        if confidence < _LOW_CONFIDENCE_FLOOR:
192            return ValidationDisposition.ESCALATE
193
194        return ValidationDisposition.CONTINUE
195
196    def finalize(self, loop_ctx: ValidationLoopContext) -> ValidationLoopResult:
197        last = loop_ctx.steps[-1] if loop_ctx.steps else None
198        obs  = last.observation if last else {}
199        return ValidationLoopResult(
200            disposition      = ValidationDisposition.FINALIZE,
201            output           = {
202                "conclusion": obs.get("conclusion", ""),
203                "confidence": last.confidence if last else 0.0,
204                "reasoning":  obs.get("reasoning", ""),
205            },
206            steps            = loop_ctx.steps,
207            iterations       = loop_ctx.iteration,
208            final_confidence = last.confidence if last else 0.0,
209            evidence         = [
210                s.observation.get("reasoning", str(s.observation)[:120])
211                for s in loop_ctx.steps
212            ],
213            remaining_steps  = loop_ctx.remaining_steps,
214            notes            = loop_ctx.notes,
215        )

221    @staticmethod
222    def _parse_llm_json(text: str) -> Dict[str, Any]:
223        """Extract a JSON object from LLM output — tolerates markdown fences."""
224        if not text:
225            return {}
226        try:
227            return json.loads(text)
228        except (json.JSONDecodeError, TypeError):
229            pass
230        match = re.search(r"\{.*\}", str(text), re.DOTALL)
231        if match:
232            try:
233                return json.loads(match.group())
234            except json.JSONDecodeError:
235                pass
236        return {}

The loop finalizes when the LLM explicitly signals the plan is complete (empty remaining_steps) or confidence reaches confidence_threshold — whichever comes first.
