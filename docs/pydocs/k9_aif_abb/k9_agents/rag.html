52    def _extract_sections(self, text: str) -> List[Dict[str, Any]]:
53        header_pattern = re.compile(r'^(#{1,4})\s+(.+)$', re.MULTILINE)
54        matches = list(header_pattern.finditer(text))
55
56        if not matches:
57            return [{"title": "Document", "level": 0, "content": text.strip()}]
58
59        sections = []
60        for i, match in enumerate(matches):
61            level = len(match.group(1))
62            title = match.group(2).strip()
63            start = match.end()
64            end = matches[i + 1].start() if i + 1 < len(matches) else len(text)
65            content = text[start:end].strip()
66            if content:
67                sections.append({"title": title, "level": level, "content": content})
68
69        return sections
70
71    def _chunk_sections(self, sections: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
72        chunks = []
73        for section in sections:
74            content = section["content"]
75            words = content.split()
76            if len(words) <= self.max_chunk_tokens:
77                chunks.append({
78                    "text": content,
79                    "section": section["title"],
80                    "index": len(chunks),
81                    "token_estimate": len(words),
82                })
83            else:
84                start = 0
85                while start < len(words):
86                    end = min(start + self.max_chunk_tokens, len(words))
87                    chunk_words = words[start:end]
88                    chunks.append({
89                        "text": " ".join(chunk_words),
90                        "section": section["title"],
91                        "index": len(chunks),
92                        "token_estimate": len(chunk_words),
93                    })
94                    start = end - self.overlap_tokens if end < len(words) else end
95
96        return chunks

23    layer = "RAG K9EmbeddingAgent OOB"

25    def __init__(self, config: Optional[Dict[str, Any]] = None, monitor=None, **kwargs):
26        super().__init__(config or {}, monitor=monitor, **kwargs)
27        self._vectordb = None
28        self._embedding_svc = None
29
30    def _ensure_vectordb(self):
31        if self._vectordb is None:
32            self._vectordb = VectorDBFactory.from_config(self.config)
33
34    def _ensure_embedding_svc(self):
35        if self._embedding_svc is None:
36            self._embedding_svc = EmbeddingServiceFactory.create(self.config)
37
38    def execute(self, payload: Dict[str, Any]) -> Dict[str, Any]:
39        chunks = payload.get("chunks", [])
40
41        if not chunks:
42            self.logger.warning("[%s] No chunks in payload", self.layer)
43            return {"agent": "K9EmbeddingAgent", "indexed": 0, "error": "no chunks"}
44
45        self._ensure_vectordb()
46        self._ensure_embedding_svc()
47
48        ids: List[str] = []
49        embeddings: List[List[float]] = []
50        documents: List[str] = []
51        metadatas: List[Dict[str, Any]] = []
52
53        for chunk in chunks:
54            text = chunk.get("text", "")
55            if not text:
56                continue
57
58            doc_id = hashlib.md5(text.encode()).hexdigest()
59            embedding = self._embedding_svc.embed(text)
60
61            if not embedding:
62                self.logger.warning("[%s] Empty embedding for chunk %s", self.layer, chunk.get("index"))
63                continue
64
65            ids.append(doc_id)
66            embeddings.append(embedding)
67            documents.append(text)
68            metadatas.append({
69                "section": chunk.get("section", ""),
70                "index": chunk.get("index", 0),
71            })
72
73        if ids:
74            self._vectordb.insert_batch(ids, embeddings, documents, metadatas)
75
76        self.publish_event({"type": "EmbeddingsGenerated", "count": len(ids)})
77        self.logger.info("[%s] Indexed %d chunks into VectorDB", self.layer, len(ids))
78
79        return {
80            "agent": "K9EmbeddingAgent",
81            "indexed": len(ids),
82            "skipped": len(chunks) - len(ids),
83        }

23    layer = "RAG K9RetrievalAgent OOB"

25    def __init__(self, config: Optional[Dict[str, Any]] = None, monitor=None, **kwargs):
26        super().__init__(config or {}, monitor=monitor, **kwargs)
27        self._vectordb = None
28        self._embedding_svc = None
29        self.top_k = self.config.get("top_k", 5)
30
31    def _ensure_vectordb(self):
32        if self._vectordb is None:
33            self._vectordb = VectorDBFactory.from_config(self.config)

35    def _ensure_embedding_svc(self):

37            self._embedding_svc = EmbeddingServiceFactory.create(self.config)
38
39    def execute(self, payload: Dict[str, Any]) -> Dict[str, Any]:
40        query = payload.get("query", "") or payload.get("question", "") or payload.get("prompt", "")
41
42        if not query:
43            self.logger.warning("[%s] No query in payload", self.layer)
44            return {"agent": "K9RetrievalAgent", "retrieved": [], "count": 0, "error": "no query"}
45
46        self._ensure_vectordb()
47        self._ensure_embedding_svc()
48
49        query_embedding = self._embedding_svc.embed(query)
50
51        if not query_embedding:
52            self.logger.error("[%s] Failed to generate query embedding", self.layer)
53            return {"agent": "K9RetrievalAgent", "retrieved": [], "count": 0, "error": "embedding failed"}
54
55        hits = self._vectordb.search(query_embedding, top_k=self.top_k)
56
57        context_text = "\n\n---\n\n".join(h.get("text", "") for h in hits if h.get("text"))
58
59        self.publish_event({"type": "RetrievalCompleted", "query": query[:100], "hits": len(hits)})
60        self.logger.info("[%s] Retrieved %d chunks for query: %s", self.layer, len(hits), query[:80])
61
62        return {
63            "agent": "K9RetrievalAgent",
64            "retrieved": hits,
65            "count": len(hits),
66            "context": context_text,
67            "query": query,
68        }
