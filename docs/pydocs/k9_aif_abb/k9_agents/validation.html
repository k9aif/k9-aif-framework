 51class BaseValidationLoopAgent(BaseAgent):
 52    """
 53    ABB: agent that iterates a hypothesis-validate-reason cycle until it can
 54    FINALIZE, ESCALATE, or FAIL with evidence-backed output.
 55
 56    Public interface is unchanged from BaseAgent — callers call
 57    ``execute(payload)`` and receive a ``dict``.  The loop is internal.
 58
 59    Subclasses must implement:
 60        generate_hypothesis()
 61        run_validation()
 62        evaluate_observation()
 63        should_continue()
 64        finalize()
 65
 66    Subclasses may override:
 67        escalate()   — default returns disposition=ESCALATE with generic output
 68        fail()       — default returns disposition=FAIL with generic output
 69    """
 70

 77    def execute(self, payload: Dict[str, Any]) -> Dict[str, Any]:
 78        max_iterations      = int(self.config.get("max_iterations", _DEFAULT_MAX_ITERATIONS))
 79        finalize_on_max     = self._parse_bool(
 80            self.config.get("finalize_on_max_iterations"), _DEFAULT_FINALIZE_ON_MAX
 81        )
 82        escalate_on_error   = self._parse_bool(
 83            self.config.get("escalate_on_tool_error"), False
 84        )
 85
 86        loop_ctx = ValidationLoopContext(payload=payload)
 87        self._emit("loop_started", {"max_iterations": max_iterations})
 88
 89        while loop_ctx.iteration < max_iterations:
 90            loop_ctx.iteration += 1
 91
 92            hypothesis = self.generate_hypothesis(loop_ctx)
 93            self._emit("hypothesis_generated", {
 94                "iteration": loop_ctx.iteration,
 95                "hypothesis": str(hypothesis),
 96            })
 97
 98            # Item 5: catch run_validation() errors — never let a tool failure
 99            # crash the loop; convert to FAIL or ESCALATE per config.
100            try:
101                tool_result = self.run_validation(hypothesis, loop_ctx)
102                self._emit("validation_tool_invoked", {"iteration": loop_ctx.iteration})
103            except Exception as exc:
104                self.logger.error(
105                    "[%s] run_validation raised at iteration %d: %s",
106                    self.layer, loop_ctx.iteration, exc,
107                )
108                error_disposition = (
109                    ValidationDisposition.ESCALATE if escalate_on_error
110                    else ValidationDisposition.FAIL
111                )
112                loop_ctx.steps.append(ValidationLoopStep(
113                    iteration   = loop_ctx.iteration,
114                    hypothesis  = hypothesis,
115                    tool_result = None,
116                    observation = {"error": str(exc), "confidence": 0.0},
117                    disposition = error_disposition,
118                    confidence  = 0.0,
119                ))
120                event = "loop_escalated" if escalate_on_error else "loop_failed"
121                self._emit(event, {"iterations": loop_ctx.iteration, "error": str(exc)})
122                handler = self.escalate if escalate_on_error else self.fail
123                return self._to_dict(handler(loop_ctx))
124
125            observation = self.evaluate_observation(tool_result, loop_ctx)
126            self._emit("observation_evaluated", {
127                "iteration": loop_ctx.iteration,
128                "observation": str(observation),
129            })
130
131            disposition = self.should_continue(observation, loop_ctx)
132            confidence  = self._extract_confidence(observation)
133
134            loop_ctx.steps.append(ValidationLoopStep(
135                iteration   = loop_ctx.iteration,
136                hypothesis  = hypothesis,
137                tool_result = tool_result,
138                observation = observation,
139                disposition = disposition,
140                confidence  = confidence,
141            ))
142
143            if disposition is ValidationDisposition.FINALIZE:
144                self._emit("loop_finalized", {"iterations": loop_ctx.iteration})
145                return self._to_dict(self.finalize(loop_ctx))
146
147            if disposition is ValidationDisposition.ESCALATE:
148                self._emit("loop_escalated", {"iterations": loop_ctx.iteration})
149                return self._to_dict(self.escalate(loop_ctx))
150
151            if disposition is ValidationDisposition.FAIL:
152                self._emit("loop_failed", {"iterations": loop_ctx.iteration})
153                return self._to_dict(self.fail(loop_ctx))
154
155            self._emit("loop_continued", {"iteration": loop_ctx.iteration})
156
157        # max iterations reached without a terminal disposition
158        if finalize_on_max:
159            self._emit("loop_finalized", {
160                "reason": "max_iterations_reached",
161                "iterations": loop_ctx.iteration,
162            })
163            return self._to_dict(self.finalize(loop_ctx))
164
165        self._emit("loop_escalated", {
166            "reason": "max_iterations_reached",
167            "iterations": loop_ctx.iteration,
168        })
169        return self._to_dict(self.escalate(loop_ctx))

254    def _emit(self, event_type: str, data: Dict[str, Any]) -> None:
255        # publish_event() is confirmed on BaseAgent. The hasattr guard protects
256        # against test doubles or future subclasses that omit the mixin.
257        event = {"type": event_type, "agent": self.layer, **data}
258        if hasattr(self, "publish_event"):
259            self.publish_event(event)
260        else:
261            self.logger.info("[%s] event: %s", self.layer, event)

263    def _extract_confidence(self, observation: Any) -> float:
264        # Clamp to [0.0, 1.0] — bad subclasses must not break confidence semantics.
265        if isinstance(observation, dict):
266            try:
267                return min(1.0, max(0.0, float(observation.get("confidence", 0.0))))
268            except (TypeError, ValueError):
269                return 0.0
270        return 0.0
271
272    def _last_confidence(self, loop_ctx: ValidationLoopContext) -> float:
273        return loop_ctx.steps[-1].confidence if loop_ctx.steps else 0.0

275    @staticmethod
276    def _parse_bool(value: Any, default: bool) -> bool:
277        """Parse a config value to bool without misreading string "false" as True."""
278        if value is None:
279            return default
280        if isinstance(value, bool):
281            return value
282        if isinstance(value, int):
283            return bool(value)
284        if isinstance(value, str):
285            return value.strip().lower() in ("true", "1", "yes", "on")
286        return default

288    def _to_dict(self, result: ValidationLoopResult) -> Dict[str, Any]:
289        return {
290            "agent":            self.layer,
291            "disposition":      result.disposition.value,
292            "output":           result.output,
293            "iterations":       result.iterations,
294            "final_confidence": result.final_confidence,
295            "evidence":         result.evidence,
296            "steps": [
297                {
298                    "iteration":   s.iteration,
299                    "hypothesis":  str(s.hypothesis),
300                    "observation": str(s.observation),
301                    "disposition": s.disposition.value,
302                    "confidence":  s.confidence,
303                    # tool_result intentionally excluded — raw tool output may
304                    # contain sensitive data. Add sanitized_tool_result in
305                    # subclass finalize() if inspection is needed.
306                }
307                for s in result.steps
308            ],
309        }

 89    def generate_hypothesis(self, loop_ctx: ValidationLoopContext) -> str:
 90        role = self.config.get("role", _DEFAULT_ROLE)
 91        goal = self.config.get("goal", _DEFAULT_GOAL)
 92
 93        prior_context = ""
 94        if loop_ctx.steps:

 96                f"  Iteration {s.iteration}: confidence={s.confidence:.2f} — "

109            f"  conclusion  : string  — your finding\n"
110            f"  confidence  : float   — 0.0 (no confidence) to 1.0 (certain)\n"

115    def run_validation(self, hypothesis: str, loop_ctx: ValidationLoopContext) -> str:
116        req = InferenceRequest(
117            prompt=hypothesis,
118            task_type=self.config.get("model", "reasoning"),
119            metadata={"agent": self.layer, "iteration": loop_ctx.iteration},
120        )
121        resp = llm_invoke(self.config, req)
122        return resp.output

124    def evaluate_observation(self, tool_result: str, loop_ctx: ValidationLoopContext) -> Dict[str, Any]:
125        data = self._parse_llm_json(tool_result)
126        return {
127            "conclusion": data.get("conclusion", str(tool_result)[:200]),
128            "confidence": data.get("confidence", 0.5),
129            "reasoning":  data.get("reasoning", ""),
130            "needs_more": bool(data.get("needs_more", False)),
131        }

133    def should_continue(self, observation: Dict[str, Any], loop_ctx: ValidationLoopContext) -> ValidationDisposition:
134        threshold  = float(self.config.get("confidence_threshold", 0.8))
135        max_iter   = int(self.config.get("max_iterations", 5))
136        confidence = observation["confidence"]
137
138        if confidence >= threshold:
139            return ValidationDisposition.FINALIZE
140
141        has_room = loop_ctx.iteration < max_iter - 1
142        if observation.get("needs_more") and has_room:
143            return ValidationDisposition.CONTINUE
144
145        if confidence < _LOW_CONFIDENCE_FLOOR:
146            return ValidationDisposition.ESCALATE
147
148        return ValidationDisposition.CONTINUE

150    def finalize(self, loop_ctx: ValidationLoopContext) -> ValidationLoopResult:
151        last = loop_ctx.steps[-1] if loop_ctx.steps else None
152        obs  = last.observation if last else {}
153        return ValidationLoopResult(
154            disposition      = ValidationDisposition.FINALIZE,
155            output           = {
156                "conclusion": obs.get("conclusion", ""),
157                "confidence": last.confidence if last else 0.0,
158                "reasoning":  obs.get("reasoning", ""),
159            },
160            steps            = loop_ctx.steps,
161            iterations       = loop_ctx.iteration,
162            final_confidence = last.confidence if last else 0.0,
163            evidence         = [
164                s.observation.get("reasoning", str(s.observation)[:120])

OOB K9 Validation Loop Agent — LLM-driven iterative reasoning.

The LLM is the validation tool. Each iteration asks the LLM to assess the payload, return a confidence score, and indicate whether further analysis would improve its answer. The loop continues until the LLM reaches confidence_threshold, exhausts max_iterations, or signals it cannot improve further.

Subclasses may override any of the five methods to customise behaviour without re-implementing the full loop. Typical pattern:
