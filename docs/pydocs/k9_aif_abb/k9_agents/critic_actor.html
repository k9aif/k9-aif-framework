 66class BaseCriticActorAgent(BaseAgent):
 67    """
 68    ABB: Actor-Critic refinement loop.
 69
 70    The Actor generates or refines a draft; the Critic evaluates it and
 71    returns structured feedback.  The loop continues until the Critic
 72    accepts the output, a terminal disposition is reached, or max_rounds
 73    is hit.
 74
 75    Public interface is unchanged from BaseAgent — callers call
 76    ``execute(payload)`` and receive a ``dict``.  The loop is internal.
 77
 78    Subclasses must implement:
 79        generate()
 80        critique()
 81        refine()
 82        should_accept()
 83        finalize()
 84
 85    Subclasses may override:
 86        escalate()  — default returns disposition=ESCALATE
 87        fail()      — default returns disposition=FAIL
 88    """

 96    def execute(self, payload: Dict[str, Any]) -> Dict[str, Any]:
 97        max_rounds        = int(self.config.get("max_rounds", _DEFAULT_MAX_ROUNDS))
 98        finalize_on_max   = self._parse_bool(
 99            self.config.get("finalize_on_max_rounds"), _DEFAULT_FINALIZE_ON_MAX
100        )
101        escalate_on_error = self._parse_bool(
102            self.config.get("escalate_on_critic_error"), False
103        )
104
105        ctx = CriticActorContext(payload=payload)
106        self._emit("loop_started", {"max_rounds": max_rounds})
107
108        draft = None
109
110        while ctx.round < max_rounds:
111            ctx.round += 1
112
113            # Actor: generate on round 1, refine on subsequent rounds
114            if ctx.round == 1:
115                draft = self.generate(ctx)
116                self._emit("draft_generated", {"round": ctx.round})
117            else:
118                prior = ctx.steps[-1]
119                draft = self.refine(prior.draft, prior.feedback, ctx)
120                self._emit("draft_refined", {"round": ctx.round})
121
122            # Critic: evaluate the draft
123            try:
124                feedback = self.critique(draft, ctx)
125                self._emit("critique_produced", {"round": ctx.round})
126            except Exception as exc:
127                self.logger.error(
128                    "[%s] critique() raised at round %d: %s",
129                    self.layer, ctx.round, exc,
130                )
131                error_disp = (
132                    CriticActorDisposition.ESCALATE if escalate_on_error
133                    else CriticActorDisposition.FAIL
134                )
135                ctx.steps.append(CriticActorStep(
136                    round=ctx.round, draft=draft,
137                    feedback={"error": str(exc)},
138                    disposition=error_disp, score=0.0,
139                ))
140                event = "loop_escalated" if escalate_on_error else "loop_failed"
141                self._emit(event, {"rounds": ctx.round, "error": str(exc)})
142                handler = self.escalate if escalate_on_error else self.fail
143                return self._to_dict(handler(ctx))
144
145            disposition = self.should_accept(feedback, ctx)
146            score = self._extract_score(feedback)
147
148            ctx.steps.append(CriticActorStep(
149                round=ctx.round, draft=draft,
150                feedback=feedback,
151                disposition=disposition, score=score,
152            ))
153
154            if disposition is CriticActorDisposition.ACCEPTED:
155                self._emit("loop_accepted", {"rounds": ctx.round})
156                return self._to_dict(self.finalize(ctx))
157
158            if disposition is CriticActorDisposition.ESCALATE:
159                self._emit("loop_escalated", {"rounds": ctx.round})
160                return self._to_dict(self.escalate(ctx))
161
162            if disposition is CriticActorDisposition.FAIL:
163                self._emit("loop_failed", {"rounds": ctx.round})
164                return self._to_dict(self.fail(ctx))
165
166            self._emit("loop_rejected", {"round": ctx.round, "score": score})
167
168        # max rounds reached without terminal disposition
169        if finalize_on_max:
170            self._emit("loop_accepted", {
171                "reason": "max_rounds_reached",
172                "rounds": ctx.round,
173            })
174            return self._to_dict(self.finalize(ctx))
175
176        self._emit("loop_escalated", {
177            "reason": "max_rounds_reached",
178            "rounds": ctx.round,
179        })
180        return self._to_dict(self.escalate(ctx))

267    def _emit(self, event_type: str, data: Dict[str, Any]) -> None:
268        event = {"type": event_type, "agent": self.layer, **data}
269        if hasattr(self, "publish_event"):

274    def _extract_score(self, feedback: Any) -> float:
275        if isinstance(feedback, dict):
276            try:
277                return min(1.0, max(0.0, float(feedback.get("score", 0.0))))
278            except (TypeError, ValueError):
279                return 0.0
280        return 0.0
281
282    def _last_score(self, ctx: CriticActorContext) -> float:
283        return ctx.steps[-1].score if ctx.steps else 0.0

285    @staticmethod
286    def _parse_bool(value: Any, default: bool) -> bool:
287        if value is None:
288            return default
289        if isinstance(value, bool):
290            return value
291        if isinstance(value, int):
292            return bool(value)
293        if isinstance(value, str):
294            return value.strip().lower() in ("true", "1", "yes", "on")
295        return default

297    def _to_dict(self, result: CriticActorResult) -> Dict[str, Any]:
298        return {
299            "agent":       self.layer,
300            "disposition": result.disposition.value,
301            "output":      result.output,
302            "rounds":      result.rounds,
303            "final_score": result.final_score,
304            "critique_log": result.critique_log,
305            "steps": [
306                {
307                    "round":       s.round,
308                    "disposition": s.disposition.value,
309                    "score":       s.score,
310                    "issues":      s.feedback.get("issues", []),
311                    # draft and full feedback intentionally excluded —
312                    # may contain sensitive content; add in finalize() if needed
313                }
314                for s in result.steps
315            ],
316        }

126    def refine(self, draft: str, feedback: Dict[str, Any], ctx: CriticActorContext) -> str:
127        role   = self.config.get("role", _DEFAULT_ACTOR_ROLE)
128        goal   = self.config.get("goal", _DEFAULT_ACTOR_GOAL)
129        issues = feedback.get("issues", [])
130
131        issue_text = "\n".join(f"  - {i}" for i in issues) if issues else "  (none specified)"
132

134            f"Role: {role}\n"
135            f"Goal: {goal}\n\n"
136            f"Original input:\n{json.dumps(ctx.payload, indent=2)}\n\n"
137            f"Previous draft (round {ctx.round - 1}):\n{draft}\n\n"
138            f"Critic feedback — issues to fix:\n{issue_text}\n\n"
139            f"Produce a revised draft that addresses all issues."
140        )
141        return self._actor_invoke(prompt, ctx)

143    def should_accept(
144        self, feedback: Dict[str, Any], ctx: CriticActorContext
145    ) -> CriticActorDisposition:
146        threshold = float(self.config.get("acceptance_threshold", 0.8))
147        score     = float(feedback.get("score", 0.0))
148        accepted  = bool(feedback.get("accepted", False))
149
150        if accepted or score >= threshold:
151            return CriticActorDisposition.ACCEPTED
152
153        # Score below 0.2 with no hope of fixing → FAIL
154        issues = feedback.get("issues", [])
155        if score < 0.2 and not issues:
156            return CriticActorDisposition.FAIL
157
158        return CriticActorDisposition.REJECTED

201    def _parse_llm_json(text: str) -> Dict[str, Any]:
202        """Extract a JSON object from LLM output — tolerates markdown fences."""
203        if not text:
204            return {}
205        try:
206            return json.loads(text)
207        except (json.JSONDecodeError, TypeError):
208            pass
209        match = re.search(r"\{[^{}]+\}", str(text), re.DOTALL)
210        if match:
211            try:
212                return json.loads(match.group())
213            except json.JSONDecodeError:
214                pass
215        return {}

143    def should_accept(
144        self, feedback: Dict[str, Any], ctx: CriticActorContext
